M0 to M9 meeting progress

Author

kd

1 Description

Creation de listes regroupant la totalité des personnes attendues comme devant être rencontrées, leur statut au jour du traitement, et certaines informations utiles supplémentaires.

Ce script s’appuie sur les résultats de :

  • La liste de l’ensemble des personnes extraites des fichiers entrées mis à disposition par la DAP depuis le début de l’étude

  • Les résultats des saisies déjà réalisées par les enquêteurs/rices et telechargées depuis le site goupile

2.0.1 M0 list creation

labeled_m0

people_to_meet_m0 = function(all_to_meet_list, labeled_m0, sorties){

    # Préparation fichiers initiaux
  people_to_meet = all_to_meet_list %>% unite("derniere_localisation", cellule, starts_with("Zone_niveau_"), sep = " > ", na.rm = TRUE, remove = TRUE)

  people_met = labeled_m0 %>% select(etab, ecrou_init,consent, consent_X__CTIME, exclus_enq,sd_603, comment) %>% mutate(consent_X__CTIME = substr(consent_X__CTIME,1,10))
  
  people_gone = sorties %>% select(ecrou_init, libelle_court_etablissement_mouvement,type_mouvement, date_et_heure_du_mouvement) %>% 
    mutate(ecrou_init = as.character(ecrou_init))
  
  #Aggregation des informations
  temp_to_meet = people_to_meet %>% 
    left_join(people_met,by = c("n_ecrou_initial" = "ecrou_init", "etablissement" = "etab")) %>% 
    left_join(people_gone, c("n_ecrou_initial" = "ecrou_init","etablissement" = "libelle_court_etablissement_mouvement"))
  
  # Ajout des labels et tris pour faciliter la lecutre
  people_to_meet_m0 = temp_to_meet %>% mutate(
    date_verif = Sys.Date(),
    jour_restant = ifelse(is.na(consent),as.Date(planifier_avant_le) - date_verif,NA),
    derniere_localisation = ifelse(is.na(consent),derniere_localisation,NA),
    Statut_M0 = case_when(
      !is.na(consent) ~ "OK",
      !is.na(date_et_heure_du_mouvement) ~ "A quitté l'étab",
      as.Date(planifier_avant_le) < date_verif ~ "Hors délais",
      as.Date(planifier_avant_le) == date_verif ~ "Dernier jour",
      (as.Date(planifier_avant_le) - date_verif) < 7  ~ "Derniere semaine",
      as.Date(planifier_a_partir_du) > date_verif ~ "Trop tôt",
      TRUE ~ NA
    )
  ) %>% select(etablissement, Statut_M0, jour_restant, everything(),-derniere_localisation, -exclus_enq, derniere_localisation, exclus_enq) %>% arrange(etablissement,jour_restant) %>% mutate(across(everything(), as.character)) %>% rename("Condamné" = "sd_603")
  
  return(people_to_meet_m0)
}

2.0.2 M3 list creation

people_to_meet_m3 = function(m0_progress){

  people_to_meet_m3 = m0_progress %>% 
    filter(consent =="Accord" & 
             is.na(type_mouvement) & 
             !(exclus_enq %in% c(1,"Oui","1"))) %>% 
  mutate(
    date_rencontre_m0 = consent_X__CTIME,
    planifier_a_partir_du = as.Date(date_entree) + 61,
    planifier_avant_le= as.Date(date_entree) + 122,
    jour_restant = as.Date(planifier_avant_le) - as.Date(date_verif),
    Statut_M3 = case_when(
      as.Date(planifier_avant_le) < as.Date(date_verif) ~ "Hors délais",
      as.Date(planifier_a_partir_du) > as.Date(date_verif) ~ "Trop tôt",
      as.Date(date_verif) >= as.Date(date_entree) + 76 &  as.Date(date_verif) <= as.Date(date_entree) + 114~ "Planification cible idéale",
      as.Date(planifier_avant_le) == as.Date(date_verif) ~ "Dernier jour",
      (as.Date(planifier_avant_le) - as.Date(date_verif)) < 15  ~ "Derniere quinzaine",
      as.Date(date_verif) < as.Date(date_entree) + 76  ~ "Planifiable (avance)",
      as.Date(date_verif) < as.Date(date_entree) + 114  ~ "Planifiable (retard)",
      TRUE ~ "Planifiable"
  )) %>% 
    select(-Statut_M0,-date_liberation_previsionnelle,-consent,-consent_X__CTIME,-type_mouvement, -derniere_localisation, -date_et_heure_du_mouvement) %>% 
  select(etablissement, Statut_M3, everything()) %>% mutate(across(everything(), as.character))
  
  return(people_to_meet_m3)
}

2.1 Progress M0 to M9 export

export_progress_list = function(m0_progress,m3_progress,m9_progress){

  investig_path = "../Equipe enquête/Ressources enquête/Listes hebdomadaires entretiens/Suivi_global/"
  coordination_path = "data/output/coordination_tools/"

  
  liste_coord <- list("M0" = m0_progress,
                      "M3" = m3_progress,
                      "M9" = m9_progress
                      )
    write.xlsx(liste_coord,paste0(coordination_path,Sys.Date(),"_m0_to_m9_progress.xlsx"),colNames = TRUE, firstRow =TRUE,borders = "rows",bandedRows=TRUE,asTable = TRUE)
  
  
  etab_list = unique(m0_progress$etablissement)
  for (etab in etab_list){
    temp_m0 = m0_progress %>% filter(etablissement == etab)
    temp_m3 = m3_progress %>% filter(etablissement == etab)
    temp_m9 = m9_progress %>% filter(etablissement == etab)

    period = c("m0","m3","m9")
    for (wave in period){
      print(paste0(investig_path, wave, "/"))
    }
    
    liste_investig <- list("M0" = temp_m0,
                      "M3" = temp_m3,
                      "M9" = temp_m9
                      )
    write.xlsx(liste_investig,paste0(investig_path, etab,"_liste_suivi_general.xlsx"),colNames = TRUE, firstRow =TRUE,borders = "rows",bandedRows=TRUE,asTable = TRUE)
  }
}
