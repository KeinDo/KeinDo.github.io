Abstract

This program purpose is DLing epsylon study wave M0, M3, and M9 results. DB files are transformed to xlsx database, including raw DB version, labelled DB version, and skim for both version. Based on these DB and entrance list generated via code_entree.qmd script, program also create and export followup lists to investigator destination.

# Login/out session for each loop occurence isn't the most efficient, but no other choice to change download directory without going on non supported experimental options.

# The scrapping has to be replaced when goupile will include an API

for site in site_list:
  path_temp = os.path.join(path, site)
  os.mkdir(path_temp)
  path_temp = os.path.join(path_temp, "DB")
  os.mkdir(path_temp)
  print("\nDownload " + site +" : ")
  
  # Launch chrome + options
  options = webdriver.ChromeOptions()
  options.add_argument("--headless=new") # Trigger/disable visual navigator
  prefs = {"download.default_directory": os.path.abspath(path_temp), "download.directory_upgrade": True}
  print(os.path.abspath(path_temp))
  options.add_experimental_option("prefs", prefs)
  driver = webdriver.Chrome(options=options)
  driver.implicitly_wait(3) # seconds - absolutly needed for login and DL

  # Go to goupile login page
  driver.get(login_url+'epsylon')

  # Username and password clean then fill
  username_field = driver.find_element(By.ID, 'fm_var_0_username')
  username_field.clear()
  username_field.send_keys(username)

  password_field = driver.find_element(By.ID, 'fm_var_0_password')
  password_field.clear()
  password_field.send_keys(password)

  password_field.send_keys(Keys.RETURN) # Cred. submission

  # DL all listed study centers
  for center in center_list:
    time.sleep(dl_time) # Timer to let download time
    center_url = login_url+site+'/'+center+'/main/consent'
    dl_url = login_url+site+'/'+center+'/api/records/export'
    print("Loading || "+center+" : "+dl_url)
    driver.get(dl_url)
    print("Done || "+center+" : "+dl_url)

  time.sleep(dl_time) # DL time needed for last loop before closing connection
  driver.quit()
