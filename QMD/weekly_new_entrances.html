2 Traitement des sorties

Dans le cadre du traitement des listes d’entrants et des personnes toujours en attente d’être rencontrées des semaines précédentes, on importe les données de l’ensemble des sorties (libération, transferts, etc) repérées depuis le début de l’étude. Ces données sont retraitées pour matcher plus directement avec les autres traitements liés (voir le tar_visnetwork() )

# Fonction de traitement du fichier des sorties depuis le début de l'étude
# Le fichier zip contient plusieurs fichier distincts a bien repérer

liberation_histo = function(zip_path){
  file_list <- unzip(zip_path, list = TRUE)
  file_sortie <- file_list$Name[grep("^EPSYLON_V1", file_list$Name)]
  sorties = read_excel(unzip(zip_path, file_sortie), sheet = "Rapport 1") %>%
    clean_names() %>%
    mutate(libelle_court_etablissement_mouvement = ifelse(libelle_court_etablissement_mouvement == "CP FLEURY MEROGIS","MA FLEURY MEROGIS",libelle_court_etablissement_mouvement)) %>% 
    mutate(ecrou_init = 
             case_when(
               libelle_court_etablissement_mouvement == "MA FLEURY MEROGIS" ~ substr(n_ecrou_initial, nchar(n_ecrou_initial)-5, nchar(n_ecrou_initial))
               ,TRUE ~ substr(n_ecrou_initial, nchar(n_ecrou_initial)-4, nchar(n_ecrou_initial)))
    )
  write.csv2(sorties,"data/source/weekly_entrances/process_results/ensemble_sorties.csv") #Export pour trace
  if (file.exists(file_sortie)) {file.remove(file_sortie)}
  return(sorties)
}

3 Traitement des entrées

Le fichier des entrées se compose de 3 onglets distincts :

  1. La liste des personnes arrivées sur les 8 derniers jours glissants

  2. La liste des personnes du 1 déjà sorties de l’établissement

  3. La localisation connue au moment de l’extraction de chaque personne de 1

Le second onglet a été rendu inutile par l’ajout de l’ensemble des sorties depuis le début du projet, on peut donc directement travailler à partir de la liste des sorties, et des onglets 1 et 3 du fichier d’entrée.

3.1 Fonction d’import des rapports 1-3

rapp_import = function(zip_path, rapp_name){
  file_list <- unzip(zip_path, list = TRUE)
  filename <- file_list$Name[grep("^EPSYLON_V0_", file_list$Name)]
  rapp = read_excel(unzip(zip_path, filename), sheet = rapp_name)
  if (file.exists(filename)) {file.remove(filename)}
  return(rapp)
}

3.2 Nettoyage des rapport 1 et 3

Pour le rapport 1 on reformate les noms et la colonne des dates d’arrivée

rapp1_clean = function(rapp1){
  temp = rapp1 %>% clean_names() %>% 
    mutate(libelle_court_etablissement_mouvement = ifelse(libelle_court_etablissement_mouvement == "CP FLEURY MEROGIS","MA FLEURY MEROGIS",libelle_court_etablissement_mouvement)) %>%

  return(temp)
}

Le rapport 3 est délivré dans un format csv US au sein d’un fichier excel, qui necessite plus d’ajustement pour reconstruire la table proprement

rapp3_clean = function(rapp3){
  number_of_col = 7 # Nombre de colonnes de la table
  new_column_names = paste("Part", 1:number_of_col, sep = "") # creates new col names (going to be deleted anyway)
  
  # Spliting col adn removing starting col
  temp = separate(rapp3, names(rapp3)[1], into = new_column_names, sep = ";", remove = TRUE)
  
  colnames(temp) <- as.character(unlist(temp[1, ])) # Rename columns with 1st row
  temp <- temp[-1, ]  # Remove 1st row
  
  temp = temp %>% clean_names() %>% select(nm_ecrou_init, id_ugc, libelle, ugcmere_fk)
  
  return(temp)
}

3.3 Production liste nettoyée des entrées de la semaine à rencontrer

Pour rappel dans le cadre du protocole plusieurs restrictions s’appliquent sur les entrées qui nous interesse d’enquêter. On supprime :

  • les doublons de ligne du rapp1 socle en conservant derniere date de lib previsionnelle

  • les personnes ayant déjà au moins 1 mouvement de sortie

  • les personnes avec un statut non hebergé

  • les personnes sortant du périmetre d’enquête selon leur sexe (H/F selon l’étab)

  • Les personne entrant sur une semaine où les inclusions n’ont pas lieu pour Seysses, et Fleury

3.3.1 Liste socle

Cette liste socle est transformée à partir des différentes sources de données disponible, et ira alimenter la liste des personnes deja extraites depuis le début de l’étude, ainsi que la fonction de mise en forme destinées à préparer les listes hebdomadaire par établissement à destination des enquêteurs

weekEntrances = function(entree_clean,sorties,cell_loca_clean,cell_nomenc,calend,past_entrances){
  # Suppression des doublons en ne gardant que la dernière date de sortie prev
  weekEntrances <- entree_clean %>%
    group_by(n_ecrou_initial) %>%
    arrange(desc(date_liberation_previsionnelle)) %>%  # Triez les dates de fin par ordre décroissant
    slice(1) %>% 
  # Suppression des personnes avec un mouvement de sortie dans les jours entre leur entrées et l'extraction, avec statut d'hebergement, et déjà non considéré dans une extraction antérieure
    filter(!n_ecrou_initial %in% sorties$n_ecrou_initial &statut_hebergement == "Hébergé" & age >= 18) %>% select(-age) %>%
  
  # Nettoyage sur condition sexe selon les etab (on explicite tout pour contourner les potentielles erreur sur certains etab)
    filter(
      !(libelle_court_etablissement_mouvement == "MA STRASBOURG" & sexe == "M") &
      !(libelle_court_etablissement_mouvement == "CP LILLE LOOS SEQUEDIN" & sexe == "M") &
      !(libelle_court_etablissement_mouvement == "MA VERSAILLES" & sexe == "M") &
      !(libelle_court_etablissement_mouvement == "MA MONTAUBAN" & sexe == "F") &
      !(libelle_court_etablissement_mouvement == "CP LILLE ANNOEULLIN" & sexe == "F")
    )  %>% 
    filter(!(libelle_court_etablissement_mouvement %in% c("CP TOULOUSE SEYSSES","MA FLEURY MEROGIS") & sexe == "M" & as.Date(date_mouvement) %in% as.Date(calend$date))) %>% 
    left_join(cell_loca_clean, by = c("n_ecrou_initial" = "nm_ecrou_init")) %>% rename(ugc_cellule = id_ugc, id_ugc = ugcmere_fk) %>% 
    left_join(cell_nomenc, by = c("id_ugc" = "id_ugc_cell"))  
  weekEntrances = weekEntrances %>% filter(!n_ecrou_initial %in% past_entrances$n_ecrou_initial)
  
  return(weekEntrances)
}

full_to_meet_list_update = function(past_entrances,weekEntrances_list,entrance_zip){
  if(nrow(weekEntrances_list) > 0){
    temp = weekEntrances_list %>% mutate(src_data = basename(entrance_zip)) %>% mutate(across(everything(), as.character))
    temp = temp %>% mutate(across(where(~ !inherits(.x, "Date")), ~na_if(.x, "")))
    temp = rbind(past_entrances,temp) %>% unique()
    length(unique(temp$n_ecrou_initial))
    write.csv2(temp,paste0("data/source/weekly_entrances/process_results/already_processed/",as.Date(Sys.time()),"_save_already_processed_entrance.csv"),row.names = FALSE,)
  return(temp)
  } else(return(data.frame()))
}

investigator_week_list_prep = function(weekEntrances_list){
  temp <- weekEntrances_list %>%
    rename(cellule = libelle, 
           date_entree = date_mouvement,
           etablissement =libelle_court_etablissement_mouvement) %>% 
    arrange(etablissement, sexe,date_entree, date_liberation_previsionnelle) %>%
    mutate(n_ecrou_initial = case_when(
      etablissement == "MA FLEURY MEROGIS" ~ substr(n_ecrou_initial, nchar(n_ecrou_initial)-5, nchar(n_ecrou_initial))
        ,TRUE ~ substr(n_ecrou_initial, nchar(n_ecrou_initial)-4, nchar(n_ecrou_initial))),
           planifier_a_partir_du = format(date_entree + 7, format = "%d/%m/%Y"),
           planifier_avant_le = format(date_entree + 21, format = "%d/%m/%Y"),
           date_entree = format(date_entree, format = "%d/%m/%Y"),
           date_liberation_previsionnelle = format(date_liberation_previsionnelle, format = "%d/%m/%Y")
           ) %>% 
    select(-cp_a_lecrou, -type_mouvement, -ugc_cellule, -id_ugc, -statut_hebergement) %>%
    select(etablissement, n_ecrou_initial, sexe, date_entree, planifier_a_partir_du,  planifier_avant_le, everything())
  return(temp)
}

4.2 Export par établissement

week_list_export = function(meet_list_prep){
  bu_folder = "data/source/weekly_entrances/process_results/listes_hebdomadaires_backup/"
  if(nrow(meet_list_prep)>0){
    etab_list = unique(meet_list_prep$etablissement)
    date_extraction = as.Date(Sys.time())
    
    week_folder = paste0("Ressources enquête/Listes hebdomadaires entretiens/",date_extraction,"_liste_etab/")
  
    if (!dir.exists(week_folder)) {
      dir.create(week_folder)
    }
    for (etab in etab_list){
      to_meet_detai = meet_list_prep %>% filter(etablissement == etab)
      write.xlsx(to_meet_detai,paste0(week_folder,as.Date(Sys.time()),"_",etab,"_liste_planification.xlsx"))
    }
    file.copy(from = week_folder, to = bu_folder, recursive = TRUE)
  }
  return(bu_folder)
}

Creation d’une liste unique de toutes les personnes intégrée à une liste hebdomadaire

all_to_meet_list_func = function(bu_folder){
  to_merge = list.files(path = bu_folder, pattern = "\\planification.xlsx$", full.names = TRUE, recursive = TRUE)
  
  data_frames <- lapply(to_merge, read_excel)
  
  data_frames = lapply(data_frames, function(df) {
    if("date_entree" %in% names(df)) {
      df$date_entree <- as.Date(df$date_entree, format = "%d/%m/%Y")
    }
    if("date_liberation_previsionelle" %in% names(df)) {
      df$date_liberation_previsionelle <- as.Date(df$date_liberation_previsionelle, format = "%d/%m/%Y")
    }
    # Return the modified data frame
    return(df)
  })
  
  merged_data <- do.call(rbind, data_frames) %>% arrange(etablissement,sexe,date_entree) %>% unique() %>% 
    mutate(
    planifier_a_partir_du = as.Date(date_entree) + 7    
    ,planifier_avant_le = as.Date(date_entree) + 21
  )
  
  write.csv2(merged_data,paste0(bu_folder,"../ensemble_inclusions.csv"))
  return(merged_data)
}
