0.1 Création d’une base unique de suivi M0

On reprend une fusion de l’attendu et du réalisé qu’au augmente au cas par cas des variables necessaires aux tdb de suivi. Cette table est une alternative plus détaillée de la table m0_progress.

stat_m0_db_func = function(all_to_meet_list, labeled_m0){
  temp = all_to_meet_list %>% left_join(labeled_m0, by = c("n_ecrou_initial" = "ecrou_init")) %>% 
    mutate(di = case_when(etablissement == "CP LILLE ANNOEULLIN" ~ "Lille",
      etablissement == "CP LILLE LOOS SEQUEDIN" ~ "Lille",
      etablissement == "CP TOULOUSE SEYSSES" ~ "Toulouse",
      etablissement == "MA FLEURY MEROGIS"  ~ "Paris",
      etablissement == "MA MONTAUBAN" ~ "Toulouse",
      etablissement == "MA STRASBOURG" ~ "Strasbourg",
      etablissement == "MA VERSAILLES" ~ "Paris",
      TRUE ~ NA)) %>% 
    select(di, everything())
    return(temp)
}

0.2 Fonction de jauge

gauge_func = function(stat_bd, min_target, max_target, sex){
  value = stat_bd %>% filter(sexe == sex &
                                 consent == "Accord" &
                                 m0_real %in% c("1","Oui")) %>% nrow()

  delta_value = stat_bd %>% filter(sexe == sex &
                                     consent == "Accord" &
                                     m0_real %in% c("1","Oui") &
                                     consent_X__CTIME <= Sys.Date()-7) %>% nrow()
  day_since_begining = as.numeric(Sys.Date() - as.Date("2024-01-08"))
  total_duration = ifelse(sex == "M", round(6*30.5),round(8*30.5))

  projected_value = round(value/day_since_begining*total_duration)

  fig <- plot_ly(
    type = "indicator",
    mode = "gauge+number+delta",
    value = value,
    delta = list(reference = delta_value, increasing = list(color = "lightgreen")),
    gauge = list(
      axis = list(range = list(NULL, max_target), tickwidth = 1, tickcolor = "white"),
      bar = list(color = "darkblue"),
      bgcolor = "white",
      borderwidth = 2,
      bordercolor = "gray",
      steps = list(
        list(range = c(0, min_target), color = "lightgrey"),
        list(range = c(min_target, max_target), color = "gray")),
        threshold = list(
          line = list(color = "darkred", width = 2),
          thickness = 0.75,
          value = projected_value)
  ))

  fig <- fig %>%
    layout(
      margin = list(l=20,r=30),
      paper_bgcolor = "#343A40",
      font = list(color = "white", family = "Arial"))
}

0.3 Fonction sunburst

sunburst_func = function(stat_db, sex){
  # Data prep
  stat_db = stat_db %>% filter(sexe == sex)
  all  = stat_db %>% 
    group_by(consent) %>% summarise(values=n()) %>% 
    mutate(parents = "   ",
           consent = case_when(
             is.na(consent) ~ "En attente",
             TRUE ~ consent
                               )
           )
  
  no_inclus = stat_db %>% filter(consent == "Impossibilité") %>% 
    group_by(consent_impossible) %>% summarise(values=n()) %>% 
    mutate(parents = "Impossibilité") %>% rename(consent= consent_impossible)
  
  inclus = stat_db %>% filter(consent == "Accord") %>% 
    group_by(m0_real) %>% summarise(values=n()) %>% 
    mutate(parents = "Accord") %>% rename(consent= m0_real) %>% 
    mutate(consent= case_when(consent == 0 ~ "Interrompu",
                consent == 1 ~ "Complet",
                TRUE ~ NA
    ))
  
  
  temp = all %>% rbind(no_inclus) %>% rbind(inclus)
  
  # Figure execution
  couleurs_par_modalite <- setNames(
    c("lightgray", "#f39c12", "#6f6f6f", "#e74c3c", "#00cc96", "#434343"),
    c('Hors délai', 'En attente', 'Impossibilité', 'Refus','Accord', 'Nouvelles entrées'))
  
  fig = plot_ly(data = temp, 
                 type = "sunburst", 
                 ids = ~consent, 
                 labels = ~consent, 
                 parents = ~parents, 
                 values = ~values, 
                 branchvalues = 'total',
                 textinfo = 'label+value', # Afficher les étiquettes et les valeurs
                 # insidetextorientation = 'radial', # Orientation du texte
                 outsidetextfont = list(size = 20),
                 marker = list(
                   colors = sapply(temp$consent, function(x) couleurs_par_modalite[x]),
                   line = list(width = 2)
                               ),
                 leaf = list(opacity = 0.8)) %>%
  layout(paper_bgcolor = '#343A40')

  return(fig)
}

0.4 Fonction courbe rythme d’inclusion

inclusion_curve = function(stat_db, sex){
  # Creating matching datafr
  start_date = as.Date("2024-01-08")
  end_date= as.Date(Sys.Date())
  date = seq(from = start_date, to = end_date, by = "day")
  inclusion = as_tibble(date) %>% rename(date = value) %>% mutate(date = as.character(date))

  stat_db = stat_db %>% filter(sexe == sex)
  stat_db$date_integr_base = as.character(as.Date(stat_db$consent_X__CTIME))
  stat_db$date_integr_ajust = as.character(as.Date(stat_db$consent_X__CTIME) - 8)


  # Adding entrances count
  entrance_num = stat_db %>% group_by(date_entree) %>% summarise(count=n()) %>% arrange(date_entree) %>%
    mutate( date_entree = as.character(date_entree),
      nb_entree = cumsum(count)) %>%  select(-count)

  inclusion = inclusion %>% left_join(entrance_num, by = c("date"="date_entree"))

  vu = stat_db %>% filter(!is.na(consent)) %>%  group_by(date_integr_base) %>% summarise(count=n()) %>% mutate(nb_vu = cumsum(count)) %>% select(-count)

  vu_ajust = stat_db %>% filter(!is.na(consent)) %>%  group_by(date_integr_ajust) %>% summarise(count=n()) %>% mutate(nb_vu_ajust = cumsum(count)) %>% select(-count)

  inclusion = inclusion %>% left_join(vu, by = c("date"="date_integr_base")) %>% left_join(vu_ajust, by = c("date"="date_integr_ajust"))

  inclusion = inclusion %>% 
    mutate(
      nb_vu_ajust = if_else(is.na(nb_vu_ajust), NA_real_, nb_vu_ajust),
      nb_vu = if_else(is.na(nb_vu), NA_real_, nb_vu),
      nb_entree = if_else(is.na(nb_entree), NA_real_, nb_entree)
    ) %>% 
    fill(nb_vu,nb_entree,nb_vu_ajust, .direction = "down")

  start_date <- as.Date("2024-01-07")
  end_date <- as.Date(max(inclusion$date, na.rm = TRUE))
  date_breaks <- seq(from = start_date, to = end_date, by = "7 days")
  date_liste = seq(from = as.Date("2024-01-10"), to = end_date, by = "7 days")
  new_list <- as.data.frame(date_liste)
  new_list$ylvl = 0

  graph = ggplot(data = inclusion, aes(x = as.Date(date))) +
    geom_line(aes(y = nb_entree, color = "Nombre d'entrées cumulé",group = 1),lwd = 1) +
    geom_line(aes(y = nb_vu, color = "Nombre d'enregistrements cumulé",group = 1),lwd = 0.7) +
    geom_line(aes(y = nb_vu_ajust, color = "Nombre d'enregistrements cumulé ajusté (-8j)",group = 1),lwd = 0.4) +
    labs(x = "Date",
         y = "Effectif",
         color = "") +
    scale_color_manual(values = c("Nombre d'entrées cumulé" = "black", "Nombre d'enregistrements cumulé" = "darkblue", "Nombre d'enregistrements cumulé ajusté (-8j)" = "darkgrey")) +
    theme_dark()+
    theme(plot.background = element_rect(fill = "#343A40", color = "#343A40"),
          panel.background = element_rect(fill = "#343A40", color = "#343A40"),
          legend.background = element_rect(fill = "#343A40", color = "#343A40"),
          legend.text = element_text(color = "white"),
          legend.title = element_text(color = "white"),
          axis.text.x = element_text(angle = 90, vjust = -0.5, hjust = 1, color = "white"),
          axis.text.y = element_text(color = "white"),
          axis.title.x = element_text(color = "white"),
          axis.title.y = element_text(color = "white"),
          legend.position = "bottom") +
    ylim(0, NA)+
    theme(axis.text.x = element_text(angle = 90, vjust = -0.5, hjust = 1),
          legend.position = "bottom")+
    scale_x_date(date_breaks = "1 day", date_labels = "%Y-%m-%d") +
    geom_vline(xintercept = as.numeric(date_breaks), linetype = "dashed", color = "white", linewidth = 0.7) +
    geom_point(data = new_list, aes(y = ylvl, x = date_liste),color="black")

  return(graph)
}

graph_return = function(stat_db,di_f= ""){
  if(di_f!=""){
    stat_db = stat_db %>% filter(di == di_f)
  } else {stat_db = stat_db}

  # stat_db = stat_m0_db
  # Données valueboxes
  nb_entree = nrow(stat_db)
  nb_accord = stat_db %>% filter(consent == "Accord") %>% nrow()
  nb_refus = stat_db %>% filter(consent == "Refus") %>% nrow()
  nb_imposs = stat_db %>% filter(consent == "Impossibilité") %>% nrow()
  nb_HD = stat_db %>% filter(consent == "Hors délai") %>% nrow()
  nb_attente = stat_db %>% filter(is.na(consent)) %>% nrow()
  
  # Jauge inclusion H/F
  gauge_h = gauge_func(stat_db,800,1000, "M")
  gauge_f = gauge_func(stat_db,200,300, "F")

  # Sunburst
  if(di_f == "Strasbourg"){
    sunb_h = NULL
  } else{sunb_h = sunburst_func(stat_db,"M")}
  sunb_f = sunburst_func(stat_db,"F")
  
  #Rythme inclusion
  rythm_h = inclusion_curve(stat_db,"M")
  rythm_f = inclusion_curve(stat_db,"F")
  
  graph = list(
    nb_entree = nb_entree,
    nb_accord = nb_accord,
    nb_refus = nb_refus,
    nb_imposs = nb_imposs,
    nb_HD = nb_HD,
    nb_attente = nb_attente,
    sunb_h = sunb_h,
    sunb_f = sunb_f,
    gauge_h = gauge_h,
    gauge_f = gauge_f,
    rythm_f= rythm_f,
    rythm_h = rythm_h
  )
  return(graph)
}
